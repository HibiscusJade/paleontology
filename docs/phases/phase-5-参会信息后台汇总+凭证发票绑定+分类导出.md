### 3. AdminContext.tsx —— 导出函数

```ts
// 导出 ZIP 结构
// export_branch_{branchId}_{date}.zip
// ├── 学生会员/
// │   ├── 缴费凭证/
// │   │   ├── 张三_学生会员_2026-06-15_TRX001.pdf
// │   │   └── ...
// │   └── 电子发票/
// │       ├── 张三_学生会员_2026-06-15_INV001.pdf
// │       └── ...
// ├── 非学生会员/
// │   ├── 缴费凭证/
// │   └── 电子发票/
// ├── 学生（非会员）/
// │   ├── 缴费凭证/
// │   └── 电子发票/
// └── 非学生（非会员）/
//     ├── 缴费凭证/
//     └── 电子发票/
// └── 汇总台账.xlsx（可选，CSV 作为轻量替代）

interface ExportOptions {
  scope: "branch" | "conference";
  scopeId: string;
  includeCategories?: ConferenceFeeType[]; // 可选仅导出某些人群
}

function generateExportZip(options: ExportOptions): Blob;
```

> **注意**：由于当前为纯前端 localStorage 架构，ZIP 生成可使用 `JSZip` 库。文件命名规范采用下划线分隔：`{姓名}_{身份}_{日期}_{流水号}.{扩展名}`。

## 用户前台修改

### 4. PersonalCenter.tsx —— 凭证发票绑定展示

在「缴费记录」标签页中增强展示：
- 每条会员费记录 → 关联凭证文件 + 发票文件
- 每条会议费记录 → 关联凭证文件 + 发票文件
- 每条记录锁定缴费时的身份标签（不受后续身份变更影响）
- 支持查看/下载已上传的凭证和发票
